🌈 디자인 토큰 파이프라인 활용

이미 구축된 tokens.figma.json → Style Dictionary → Tailwind 프리셋 파이프라인을 최대한 활용

📋 현재 토큰 파이프라인

pipeline
// package.json의 build:tokens 스크립트

tokens.figma.json
  ↓ diff-tokens    // 변경 감지
  ↓ merge-tokens   // 통합
  ↓ style-dictionary // CSS 변수 생성
  ↓ generate-tailwind-preset
  ↓ sync-manager-tokens
  ↓ validate-tokens
  ↓
tailwind.preset.generated.js
  + colors.css
  + styles.css

🔗 피그마 변수 ↔ CSS 변수

매핑 예시
// 피그마 토큰
spacing/1 → 4px
spacing/2 → 8px
spacing/4 → 16px

// CSS 변수 (자동 생성됨)
--gap/1     → 4px
--padding/4 → 16px

// 피그마 디자인에서 사용되는 토큰
--color/bg/white
--color/border/normal
--font/size/body/p2
--radius/8
토큰 동기화 팁 피그마에서 Variables가 업데이트되면, npm run build:tokens 실행만으로 CSS 변수와 Tailwind 프리셋이 자동 갱신됩니다. 컴포넌트 코드를 수정할 필요 없이 디자인 변경이 반영됩니다.

tsx
import React, { useState } from 'react';
import {
  ActionButton,
  Tab,
  Chip,
  Tag,
  TextField,
} from '@likelion-design/ui';

export const TrainingLogPage = () => {
  const [activeTab, setActiveTab] = useState('absence');
  const [selectedChip, setSelectedChip] = useState('training');

  return (
    <div className="training-log-content">
      {/* 탭 영역 */}
      <Tab.Group
        type="round"
        size="large"
        value={activeTab}
        onChange={setActiveTab}
        items={[
          { value: 'absence', label: '공결 신청' },
          { value: 'later', label: '추후 적용' },
        ]}
      />

      {/* 칩 필터 */}
      <Chip.Group
        type="solid"
        size="medium"
        value={selectedChip}
        onChange={setSelectedChip}
      >
        <Chip value="training" label="훈련 사항" />
        <Chip value="project" label="프로젝트 일정" />
      </Chip.Group>

      {/* 훈련 테이블 */}
      <table className="training-table">
        <thead>
          <tr>
            <th>교시</th>
            <th>훈련과목</th>
            <th>담당강사</th>
            <th>훈련내용</th>
          </tr>
        </thead>
        <tbody>
          <tr>
            <td>1교시</td>
            <td>React 기초</td>
            <td>
              <Tag type="weak" size="small" label="김강사" />
            </td>
            <td>컴포넌트 설계</td>
          </tr>
        </tbody>
      </table>

      {/* 입력 필드 */}
      <TextField
        type="area"
        title="지시사항"
        placeholder="지시사항을 입력하세요"
      />
      <TextField
        type="area"
        title="특이사항"
        placeholder="특이사항을 입력하세요"
      />
      <TextField
        type="area"
        title="기타사항"
        placeholder="기타사항을 입력하세요"
      />

      {/* 저장 버튼 */}
      <ActionButton
        size="large"
        color="primary"
        type="solid"
        label="훈련 일지 저장"
      />
    </div>
  );
};
