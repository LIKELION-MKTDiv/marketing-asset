/**
 * 부트캠프 프로젝트별 데이터 연동 및 정렬 통합 스크립트
 * 1. 프로젝트 키 자동 추출 (파일명 기준)
 * 2. 복합 키(user_id + 최초작성일) 기반 업데이트/적재
 * 3. A열 기준 데이터 영역 정밀 탐색
 * 4. 최초작성일 기준 오름차순 자동 정렬
 */
function updateBootcampData() {
  const FOLDER_ID = '1m2zIjgCyOxGpBpJA00nRK5wVfDStBuaJ'; // RAW 데이터 폴더 ID
  const HEADER_ROW_INDEX = 1;                         // 헤더 위치 (1행)
  const PK_1 = 'user_id';                             // 제1 식별자
  const PK_2 = '최초작성일';                           // 제2 식별자 (재지원 구분용)
  const SORT_COLUMN_NAME = '최초작성일';                // 정렬 기준 컬럼

  const ss = SpreadsheetApp.getActiveSpreadsheet();
  const ssName = ss.getName();

  // 1. 프로젝트 키워드 추출 (kdt- 로 시작하는 프로젝트 코드)
  const projectMatch = ssName.match(/kdt-[a-zA-Z0-9-]+/);
  const projectKey = projectMatch ? projectMatch[0] : "";

  // 2. 대상 시트 선정 (현재 활성화된 시트 탭 우선)
  let targetSheet = ss.getActiveSheet();
  if (!targetSheet.getName().includes(projectKey)) {
    const sheets = ss.getSheets();
    for (let s of sheets) {
      if (s.getName().includes(projectKey)) {
        targetSheet = s;
        break;
      }
    }
  }

  // 3. RAW 폴더 내 프로젝트별 최신 파일 검색
  const folder = DriveApp.getFolderById(FOLDER_ID);
  const files = folder.getFiles();
  let latestFile = null;
  let lastUpdated = 0;

  while (files.hasNext()) {
    let file = files.next();
    if (file.getName().includes(projectKey)) {
      if (file.getLastUpdated().getTime() > lastUpdated) {
        lastUpdated = file.getLastUpdated().getTime();
        latestFile = file;
      }
    }
  }

  if (!latestFile) {
    SpreadsheetApp.getUi().alert(`RAW 폴더에 '${projectKey}' 관련 최신 파일이 없습니다.`);
    return;
  }

  // 4. 데이터 로드 (CSV 및 구글 시트 형식 대응)
  let rawData;
  if (latestFile.getMimeType() === MimeType.GOOGLE_SHEETS) {
    rawData = SpreadsheetApp.open(latestFile).getSheets()[0].getDataRange().getValues();
  } else {
    rawData = Utilities.parseCsv(latestFile.getBlob().getDataAsString('UTF-8'));
  }

  const csvHeader = rawData[0].map(h => h.toString().trim());
  const csvRows = rawData.slice(1);
  const csvPk1Idx = csvHeader.indexOf(PK_1);
  const csvPk2Idx = csvHeader.indexOf(PK_2);

  if (csvPk1Idx === -1 || csvPk2Idx === -1) {
    SpreadsheetApp.getUi().alert(`RAW 파일에 필수 식별자(${PK_1}, ${PK_2})가 없습니다.`);
    return;
  }

  // 5. 현재 시트의 실제 데이터 범위 탐색 (A열 기준)
  const lastCol = targetSheet.getLastColumn();
  const aColumnValues = targetSheet.getRange("A:A").getValues();
  let actualLastRow = 1;
  for (let i = aColumnValues.length - 1; i >= 0; i--) {
    if (aColumnValues[i][0] !== "") {
      actualLastRow = i + 1;
      break;
    }
  }

  const sheetHeader = targetSheet.getRange(HEADER_ROW_INDEX, 1, 1, lastCol).getValues()[0].map(h => h.toString().trim());
  const sheetPk1Idx = sheetHeader.indexOf(PK_1);
  const sheetPk2Idx = sheetHeader.indexOf(PK_2);
  const sortColIdx = sheetHeader.indexOf(SORT_COLUMN_NAME) + 1;

  // 기존 시트 데이터 맵 생성 (복합 키 기반)
  const sheetValues = targetSheet.getRange(HEADER_ROW_INDEX, 1, actualLastRow, lastCol).getValues();
  const sheetKeyMap = {};
  for (let i = 1; i < sheetValues.length; i++) {
    const k1 = String(sheetValues[i][sheetPk1Idx]).trim();
    const k2 = String(sheetValues[i][sheetPk2Idx]).trim();
    if (k1 && k2) {
      sheetKeyMap[k1 + "_" + k2] = HEADER_ROW_INDEX + i;
    }
  }

  // 6. 업데이트 및 신규 행 분리 작업
  let updateCount = 0;
  const newRows = [];
  
  csvRows.forEach(csvRow => {
    const cK1 = String(csvRow[csvPk1Idx]).trim();
    const cK2 = String(csvRow[csvPk2Idx]).trim();
    if (!cK1 || !cK2) return;

    const compositeKey = cK1 + "_" + cK2;
    const rowValues = sheetHeader.map(h => {
      const idx = csvHeader.indexOf(h);
      return idx !== -1 ? csvRow[idx] : "";
    });

    if (sheetKeyMap[compositeKey]) {
      // 기존 지원 정보 업데이트
      targetSheet.getRange(sheetKeyMap[compositeKey], 1, 1, rowValues.length).setValues([rowValues]);
      updateCount++;
    } else {
      // 신규 지원 정보 수집
      newRows.push(rowValues);
    }
  });

  // 7. 데이터 적재 및 자동 정렬 (최초작성일 오름차순)
  if (newRows.length > 0) {
    targetSheet.getRange(actualLastRow + 1, 1, newRows.length, sheetHeader.length).setValues(newRows);
  }

  const finalLastRow = targetSheet.getLastRow();
  if (finalLastRow > HEADER_ROW_INDEX && sortColIdx > 0) {
    // 2행부터 마지막 행까지 정렬 (헤더 제외)
    const sortRange = targetSheet.getRange(HEADER_ROW_INDEX + 1, 1, finalLastRow - HEADER_ROW_INDEX, lastCol);
    sortRange.sort({column: sortColIdx, ascending: true});
  }

  SpreadsheetApp.getUi().alert(`[${targetSheet.getName()}] 동기화 완료\n- 업데이트: ${updateCount}건\n- 신규 추가: ${newRows.length}건\n- 정렬 기준: ${SORT_COLUMN_NAME}`);
}
